// Cek apakah VC++ Redistributable sudah terpasang (2015-2022)
function IsVCInstalled(): Boolean;
var
  installKey: String;
begin
  installKey := 'SOFTWARE\Microsoft\VisualStudio\14.0\VC\Runtimes\x64';
  Result := RegKeyExists(HKLM, installKey);
end;

function InitializeSetup(): Boolean;
begin
  if IsVCInstalled() then begin
    Log('VC++ 2015-2022 Redistributable found.');
    Result := True;
  end else begin
    MsgBox('Microsoft VC++ Redistributable belum ditemukan. Akan dipasang sekarang.', mbInformation, MB_OK);
    Result := True;
  end;
end;
